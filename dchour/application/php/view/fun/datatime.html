        /**
         * 根据时间戳返回星期几
         * @param $time
         * @return mixed
         */
        function weekday_by_time($time)
        {
            $weekday = array('星期日','星期一','星期二','星期三','星期四','星期五','星期六');
            return $weekday[date('w', $time)];
        }

        function weekday_by_time_str($timeStr)
        {
            $time = strtotime($timeStr);
            return weekday_by_time($time);
        }
